CO_IMAGE, CO_ADDRESS, CO_상세주소 WHERE CO_NUM = ?

공고 목록 페이지 

SELECT CO_NAME "기업명" NOTICE_TITLE "공고제목" 
ACADEMIC_INFO "학력", NOTICE_DATE "등록일", 
CLOSING_DATE "마감일"
FROM NOTICE INNER JOIN NOTICE_INFO
ON NOTICE.NOTICE_CODE=NOTICE_INFO.NOTICE_CODE;
